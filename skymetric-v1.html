#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <DHT.h>

#define I2C_ADDR 0x27
#define LCD_COLS 16
#define LCD_ROWS 2
LiquidCrystal_I2C lcd(I2C_ADDR, LCD_COLS, LCD_ROWS);

#define DHTPIN 2
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

const int buzzerPin = 8;
const float tempCritical = 35.0;
const float umidadeCritical = 75.0;
const float tempCalibration = 0.5;
const float humidadeCalibration = 2.0;
bool alertaAtivo = false;
unsigned long buzzerStartTime = 0;
bool buzzerActive = false;

void setup() {
  pinMode(buzzerPin, OUTPUT);
  lcd.init();
  lcd.backlight();
  Serial.begin(9600);

  lcd.setCursor(0, 0);
  lcd.print("    PROJETO");
  lcd.setCursor(0, 1);
  lcd.print("   SKYMETRIC");

  tone(buzzerPin, 5000);
  delay(2000);
  noTone(buzzerPin);
  lcd.clear();

  lcd.setCursor(0, 0);
  lcd.print("Temp: ");
  lcd.setCursor(0, 1);
  lcd.print("Umidade: ");
  dht.begin();
}

void alertSystem(float tempC, float humidity) {
  if (tempC >= tempCritical || humidity >= umidadeCritical) {
    if (!buzzerActive) {
      tone(buzzerPin, 5000);
      buzzerActive = true;
    }

    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Alerta! Critico");
    lcd.setCursor(0, 1);
    lcd.print("Verif. Sistema!");
    alertaAtivo = true;
  } else {
    if (alertaAtivo) {
      noTone(buzzerPin);
      buzzerActive = false;
      alertaAtivo = false;

      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Temp: ");
      lcd.setCursor(7, 0);
      lcd.print(tempC, 1);
      lcd.setCursor(0, 1);
      lcd.print("Umidade: ");
      lcd.setCursor(11, 1);
      lcd.print(humidity, 1);
    }
  }
}

void loop() {
  float tempC = dht.readTemperature() + tempCalibration;
  float humidity = dht.readHumidity() + humidadeCalibration;

  if (isnan(tempC) || isnan(humidity)) {
    lcd.setCursor(0, 0);
    lcd.print("Erro leitura DHT!");
    Serial.println("Erro leitura DHT!");
    return;
  }

  if (!alertaAtivo) {
    lcd.setCursor(7, 0);
    lcd.print(tempC, 1);
    lcd.setCursor(11, 1);
    lcd.print(humidity, 1);
  }

  Serial.print("Temp: ");
  Serial.print(tempC);
  Serial.print("C, ");
  Serial.print("Umidade: ");
  Serial.print(humidity);
  Serial.println("%");

  alertSystem(tempC, humidity);
  delay(2000);
}
